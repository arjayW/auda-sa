436  struct GroupItem : BaseItem {
437  // Construction from a previously built-up vector of pointers
438  GroupItem( BaseItemPtrs &&items_ );
439  // In-line, variadic constructor that doesn't require building a vector
440  template< typename... Args >
441  GroupItem( Args&&... args )
442  { Append( std::forward< Args >( args )... ); }
443  ~GroupItem() override;
444 
445  BaseItemPtrs items;
446 
447  private:
448  // nullary overload grounds the recursion
449  void Append() {}
450  // recursive overload
451  template< typename Arg, typename... Args >
452  void Append( Arg &&arg, Args&&... moreArgs )
453  {
454  // Dispatch one argument to the proper overload of AppendOne.
455  // std::forward preserves rvalue/lvalue distinction of the actual
456  // argument of the constructor call; that is, it inserts a
457  // std::move() if and only if the original argument is rvalue
458  AppendOne( std::forward<Arg>( arg ) );
459  // recur with the rest of the arguments
460  Append( std::forward<Args>(moreArgs)... );
461  };
462 
463  // Move one unique_ptr to an item into our array
464  void AppendOne( BaseItemPtr&& ptr );
465  // This overload allows a lambda or function pointer in the variadic
466  // argument lists without any other syntactic wrapping, and also
467  // allows implicit conversions to type Factory.
468  // (Thus, a lambda can return a unique_ptr<BaseItem> rvalue even though
469  // Factory's return type is shared_ptr, and the needed conversion is
470  // appled implicitly.)
471  void AppendOne( const Factory &factory )
472  { AppendOne( std::make_unique<ComputedItem>( factory ) ); }
473  };
