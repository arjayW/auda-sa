128  {
129  // Impose certain zoom limits

131  auto availableHeight = rectHeight - octavePadding;

133  auto minSpacePerNote =

147  int GetNoteMargin(int height) const
148  { return std::min(height / 4, (GetPitchHeight(1) + 1) / 2); }
149  int GetOctaveHeight() const { return GetPitchHeight(12) + 2; }
150  // call this once before a series of calls to IPitchToY(). It
151  // sets mBottom to offset of octave 0 so that mBottomNote
152  // is located at r.y + r.height - (GetNoteMargin() + 1 + GetPitchHeight())
153  void PrepareIPitchToY(const wxRect &r) const {
154  mBottom =
155  r.y + r.height - GetNoteMargin(r.height) - 1 - GetPitchHeight(1) +
156  (mBottomNote / 12) * GetOctaveHeight() +
157  GetNotePos(mBottomNote % 12);
158  }
159  // IPitchToY returns Y coordinate of top of pitch p
160  int IPitchToY(int p) const {
161  return mBottom - (p / 12) * GetOctaveHeight() - GetNotePos(p % 12);
162  }
163  // compute the window coordinate of the bottom of an octave: This is
164  // the bottom of the line separating B and C.
165  int GetOctaveBottom(int oct) const {
166  return IPitchToY(oct * 12) + GetPitchHeight(1) + 1;
167  }
168  // Y coordinate for given floating point pitch (rounded to int)
169  int PitchToY(double p) const {
170  return IPitchToY((int) (p + 0.5));
171  }
172  // Integer pitch corresponding to a Y coordinate
173  int YToIPitch(int y);
174  // map pitch class number (0-11) to pixel offset from bottom of octave
175  // (the bottom of the black line between B and C) to the top of the
176  // note. Note extra pixel separates B(11)/C(0) and E(4)/F(5).
177  int GetNotePos(int p) const
178  { return 1 + GetPitchHeight(p + 1) + (p > 4); }
179  // get pixel offset to top of ith black key note
180  int GetBlackPos(int i) const { return GetNotePos(i * 2 + 1 + (i > 1)); }
181  // GetWhitePos tells where to draw lines between keys as an offset from
182  // GetOctaveBottom. GetWhitePos(0) returns 1, which matches the location
183  // of the line separating B and C
184  int GetWhitePos(int i) const { return 1 + (i * GetOctaveHeight()) / 7; }

210  // channels are numbered as integers 0-15, visible channels
211  // (mVisibleChannels) is a bit set. Channels are displayed as
212  // integers 1-16.
213 
214  // Allegro's data structure does not restrict channels to 16.
215  // Since there is not way to select more than 16 channels,
216  // map all channel numbers mod 16. This will have no effect
217  // on MIDI files, but it will allow users to at least select
218  // all channels on non-MIDI event sequence data.
219 #define NUM_CHANNELS 16
220  // Bitmask with all NUM_CHANNELS bits set
221 #define ALL_CHANNELS (1 << NUM_CHANNELS) - 1
222 #define CHANNEL_BIT(c) (1 << (c % NUM_CHANNELS))
223  bool IsVisibleChan(int c) const {
224  return (mVisibleChannels & CHANNEL_BIT(c)) != 0;
225  }
226  void SetVisibleChan(int c) { mVisibleChannels |= CHANNEL_BIT(c); }
227  void ClearVisibleChan(int c) { mVisibleChannels &= ~CHANNEL_BIT(c); }
228  void ToggleVisibleChan(int c) { mVisibleChannels ^= CHANNEL_BIT(c); }
229  // Solos the given channel. If it's the only channel visible, all channels
230  // are enabled; otherwise, it is set to the only visible channel.
231  void SoloVisibleChan(int c) {
232  if (mVisibleChannels == CHANNEL_BIT(c))
233  mVisibleChannels = ALL_CHANNELS;
234  else
235  mVisibleChannels = CHANNEL_BIT(c);
236  }
