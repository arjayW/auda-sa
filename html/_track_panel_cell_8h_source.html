50  // A refinement of a given rectangle partitions it along one of its axes
51  // and associates TrackPanelNodes with the partition.
52  // The sequence of coordinates should be increasing, giving left or top
53  // coordinates of sub-rectangles.
54  // Null pointers are permitted to define empty spaces with no cell object.
55  // If the first coordinate is right of or below the rectangle boundary,
56  // then that also defines an empty space at the edge.
57  // Sub-rectangles may be defined partly or wholly out of the bounds of the
58  // given rectangle. Such portions are ignored.
59  using Child = std::pair< wxCoord, std::shared_ptr<TrackPanelNode> >;
60  using Refinement = std::vector< Child >;
