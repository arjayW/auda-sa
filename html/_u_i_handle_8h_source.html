50  // Before clicking, the handle is notified that it has been "hit"
51  // This might put the handle into its first rotated state
52  // (or last, if forward is false) or mark itself as needing a highlight.

81  // Assume previously Clicked and not yet Released or Cancelled.

84  // in which case the handle will not be invoked again.

97  // connecting and disconnecting event handlers for the menu items.

99  virtual Result Release

103  // Assume previously Clicked and not yet Released or Cancelled.
104  // Cancelled in return flags has no effect.
105  virtual Result Cancel(AudacityProject *pProject) = 0;
106 
107  // Draw extras over cells. Default does nothing.
108  // Supplies only the whole panel rectangle for now.
109  // If pass is Cells, then any drawing that extends outside the cells
110  // is later overlaid with the cell bevels and the empty background color.

112  // This is invoked on the hit test target even before it is clicked,

114  virtual void DrawExtras

156 // A frequent convenience for defining a hit test.
157 // Construct a NEW handle as if hit the first time; then either keep it, or
158 // use it to overwrite the state of a previously constructed handle that has not
159 // yet been released.
160 template<typename Subclass>
161 std::shared_ptr<Subclass> AssignUIHandlePtr
162 ( std::weak_ptr<Subclass> &holder, const std::shared_ptr<Subclass> &pNew )
163 {
164  // Either assign to a null weak_ptr, or else rewrite what the weak_ptr
165  // points at. Thus a handle already pointed at changes its state but not its
166  // identity. This may matter for the framework that holds the strong
167  // pointers.
168  auto ptr = holder.lock();
169  if (!ptr) {
170  holder = pNew;
171  return pNew;
172  }
173  else {
174  auto code = Subclass::NeedChangeHighlight( *ptr, *pNew );
175  *ptr = std::move(*pNew);
176  ptr->SetChangeHighlight( code );
177  return ptr;
178  }
179 }
